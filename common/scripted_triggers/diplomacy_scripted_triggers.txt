#	Triggers to control if diplomatic actions are enabled:
#		Format:
#
#		<diplomatic action tag>_enabled_trigger = {
#			<trigger contents>
#		}
#
#		ROOT is the country initiating the diplomatic action.
#		FROM is the recipient or target of the diplomatic action.
#
#		Note:  For cleaner tooltips, it is recommended that you use
#			if triggers within the enable triggers, and a single
#			custom_trigger_tooltip trigger within each if trigger.
#
#		Example: (Germany is prevented from declaring war on the Soviet Union.)
#
#		DIPLOMACY_WAR_ENABLE_TRIGGER = {
#			if = {
#				limit = {
#					tag = GER
#				}
#				custom_trigger_tooltip = {
#					tooltip = PREVENT_GER_WAR_ON_SOV
#					FROM = {
#						NOT = {
#							tag = SOV
#						}
#					}
#				}
#			}
#		}

DIPLOMACY_SEND_EXP_FORCE_ENABLE_TRIGGER = {
	custom_trigger_tooltip = {
		tooltip = DIPLO_DOESNT_HAVE_GBP
		OR = {
			FROM = { NOT = { has_tech = trench_warfare } }
			AND = {
				FROM = { has_tech = trench_warfare }
				ROOT = { has_tech = trench_warfare }
			}
		}
	}
}

DIPLOMACY_LEND_LEASE_ENABLE_TRIGGER = {
	custom_trigger_tooltip = {
		tooltip = DIPLO_IS_NEUTRAL_TOOLTIP
		FROM = { NOT = { is_neutral_nation = yes } }
		NOT = { is_neutral_nation = yes }
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = FREE
			}
		}
		always = yes
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = LIMITED
			}
		}
		always = yes #handled by game logic, see above
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = SAME_IDEOLOGY
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_LEND_LEASE_SAME_IDEOLOGY_TT
			has_government = FROM
		}
	}
	if = {
		limit = {
			has_game_rule = { #who tf even uses this gamerule
				rule = allow_lend_lease
				option = SAME_FACTION
			}
		}
		if = {
			limit = {
				OR = {
					AND = { # GER can LL to SPRs and FIN 
						FROM = {original_tag = GER}
						OR = {
							original_tag = SPR
							original_tag = FIN
						}
					}
					AND = { # SOV can LL to SPRs and CHI
						FROM = {original_tag = SOV}
						OR = {
							original_tag = SPR
							original_tag = CHI
						}
					}
				}
			}
			always = yes
		}
		else = {
			custom_trigger_tooltip = {
				tooltip = RULE_ALLOW_LEND_LEASE_SAME_FACTION_TT
				is_in_faction_with = FROM
			}
		}
		
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = BLOCKED
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_LEND_LEASE_BLOCKED_TT
			always = no
		}
	}
}

DIPLOMACY_SEND_VOLUNTEERS_ENABLE_TRIGGER = {
	custom_trigger_tooltip = {
		tooltip = DIPLO_HAS_SPECTATOR_IDEA
		if = {
			limit = { has_global_flag = scw_over }
			FROM = { has_idea = HMM_spectator }
		}
	}
	custom_trigger_tooltip = {
		tooltip = DIPLO_IS_NEUTRAL_TOOLTIP
		FROM = { NOT = { is_neutral_nation = yes } }
		NOT = { is_neutral_nation = yes }
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_volunteers
				option = ALWAYS_FREE
			}
		}
		if = {
			limit = {
				NOT = { tag = SPC }
				FROM = {
					tag = SPC
					has_completed_focus = SPR_regional_defense_council_of_aragon
				}
			}
			custom_trigger_tooltip = {
				tooltip = RULE_VOLUNTEERS_NOT_ANARCHISTS_TT
				NOT = { FROM = { tag = SPC } }
			}
		}
		else = {
			always = yes
		}
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_volunteers
				option = LIMITED
			}
		}
		if = { #Bulgarian volunteers in SCW (with La Resistance)
			limit = {
				has_dlc = "Battle for the Bosporus"
				tag = BUL
				FROM = { original_tag = SPR }
				has_dlc = "La Resistance"
				NOT = { has_global_flag = scw_over }
				
			}
			custom_trigger_tooltip = {
				tooltip = RULE_VOLUNTEERS_BULGARIA_SCW_TT
				NOT = {
					AND = {
						has_completed_focus = BUL_support_the_spanish_coup
						FROM = { NOT = { tag = SPA } }
					}
				}
			}
		}
		else_if = { #Bulgarian volunteers in SCW (without La Resistance)
			limit = {
				has_dlc = "Battle for the Bosporus"
				NOT = { has_dlc = "La Resistance" }
				tag = BUL
				FROM = { original_tag = SPR }	
				any_country = {
					original_tag = SPR
					has_civil_war = yes
				}
			}
			custom_trigger_tooltip = {
				tooltip = RULE_VOLUNTEERS_BULGARIA_SCW_TT
				NOT = {
					AND = {
						has_completed_focus = BUL_support_the_spanish_coup
						FROM = {
							NOT = { has_government = fascism }
							NOT = { has_government = neutrality }
						}
					}
				}
			}
		}
		else_if = { #Soviet volunteers to China (Communist Soviet -> Only to the supported China)
			limit = {
				original_tag = SOV
				has_government = communism
				FROM = { tag = CHI }
			}
			custom_trigger_tooltip = {
				tooltip = RULE_VOLUNTEERS_SOVIET_CHINA_SUPPORT_CHOSEN_ONE_TT
				always = no
				#HMMTODO fix all the diplo triggers
			}
		}
		else = {
			always = yes
		}
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_volunteers
				option = SAME_IDEOLOGY
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_VOLUNTEERS_SAME_IDEOLOGY_TT
			has_government = FROM
		}
		if = {
			limit = {
				NOT = { tag = SPC }
				FROM = {
					tag = SPC
					has_completed_focus = SPR_regional_defense_council_of_aragon
	}
			}
			custom_trigger_tooltip = {
				tooltip = RULE_VOLUNTEERS_NOT_ANARCHISTS_TT
				NOT = { FROM = { tag = SPC } }
			}
		}
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_volunteers
				option = BLOCKED
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_VOLUNTEERS_BLOCKED_TT
			always = no
		}
	}
