scripted_gui = {
	HMM_removedocksilos = {
		context_type = selected_state_context
		window_name = "remove_docksilos"
		parent_window_token = selected_state_view
        visible = {
            always = yes
        }
		effects = {
			remove_dock_click = {
				if = {
					limit = { is_fully_controlled_by = ROOT }
					remove_building = {
						type = dockyard
						level = 1
					}
				}
			}
            remove_silo_click = {
				if = {
					limit = { is_fully_controlled_by = ROOT }
					remove_building = {
						type = fuel_silo
						level = 1
					}
				}
			}
		}
		triggers = {
			remove_dock_visible = { is_fully_controlled_by = ROOT }
			remove_dock_img_visible = { is_fully_controlled_by = ROOT }
			remove_silo_visible = { is_fully_controlled_by = ROOT }
			remove_silo_img_visible = { is_fully_controlled_by = ROOT }

			remove_dock_click_enabled = {
				is_fully_controlled_by = ROOT
				dockyard > 0
			}
            remove_silo_click_enabled = {
				is_fully_controlled_by = ROOT
				fuel_silo > 0
			}
		}
	}
}
